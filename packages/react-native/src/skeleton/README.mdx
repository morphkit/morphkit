import { Skeleton } from "./Skeleton";
import { View } from "react-native";

# Skeleton

A placeholder loading component with shimmer animation.

## Basic Usage

<View style={{ padding: 16, gap: 12 }}>
  <Skeleton />
</View>

## Variants

<View style={{ padding: 16, gap: 12 }}>
  <Skeleton variant="rect" />
  <Skeleton variant="circle" />
  <Skeleton variant="text" />
</View>

## Custom Dimensions

<View style={{ padding: 16, gap: 12 }}>
  <Skeleton width={200} height={100} />
  <Skeleton variant="circle" width={60} height={60} />
  <Skeleton width="80%" height={15} />
</View>

## Content Placeholder

<View style={{ padding: 16, gap: 12 }}>
  <View style={{ flexDirection: "row", gap: 12 }}>
    <Skeleton variant="circle" width={48} height={48} />
    <View style={{ flex: 1, gap: 8 }}>
      <Skeleton variant="text" width="60%" />
      <Skeleton variant="text" width="40%" />
    </View>
  </View>
  <Skeleton width="100%" height={200} />
</View>

## Props

| Prop    | Type                           | Default   | Description            |
| ------- | ------------------------------ | --------- | ---------------------- |
| variant | "rect" \\| "circle" \\| "text" | "rect"    | Shape type             |
| width   | number \\| string              | Variant   | Custom width           |
| height  | number \\| string              | Variant   | Custom height          |
| style   | StyleProp\<ViewStyle\>         | -         | Custom container styles |

## Usage Example

```tsx
import { Skeleton } from "@warp-ui/react-native";

function LoadingCard() {
  return (
    <View style={{ padding: 16, gap: 12 }}>
      <Skeleton variant="circle" width={48} height={48} />
      <Skeleton variant="text" />
      <Skeleton variant="text" width="70%" />
      <Skeleton width="100%" height={200} />
    </View>
  );
}
```
