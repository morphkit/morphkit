# Toast

A temporary notification component with slide animation and auto-dismiss.

## Basic Usage

<View style={{ padding: 16, position: "relative", height: 120 }}>
  <Toast
    visible={true}
    message="Changes saved successfully"
    duration={0}
  />
</View>

## Variants

<View style={{ padding: 16, gap: 100, position: "relative", height: 400 }}>
  <Toast visible={true} variant="info" message="New features are available" duration={0} />
  <Toast visible={true} variant="success" message="Payment processed successfully" duration={0} />
  <Toast visible={true} variant="warning" message="Your session will expire soon" duration={0} />
  <Toast visible={true} variant="error" message="Failed to connect to server" duration={0} />
</View>

## Positions

<View style={{ padding: 16, gap: 100, position: "relative", height: 200 }}>
  <Toast visible={true} position="top" message="This toast appears at the top" duration={0} />
  <Toast visible={true} position="bottom" message="This toast appears at the bottom" duration={0} />
</View>

## Props

| Prop      | Type                                          | Default    | Description                    |
| --------- | --------------------------------------------- | ---------- | ------------------------------ |
| visible   | boolean                                       | -          | Controls visibility            |
| message   | string                                        | -          | Notification text              |
| variant   | "info" \\| "success" \\| "warning" \\| "error" | "info"     | Color theme                    |
| duration  | number                                        | 3000       | Auto-dismiss ms (0 = manual)   |
| position  | "top" \\| "bottom"                            | "bottom"   | Screen position                |
| onDismiss | () => void                                    | -          | Dismiss callback               |
| style     | StyleProp\<ViewStyle\>                        | -          | Custom container styles        |

## Usage Example

```tsx
import { Toast } from "@warp-ui/react-native";
import { useState } from "react";

function SaveButton() {
  const [toastVisible, setToastVisible] = useState(false);

  const handleSave = async () => {
    await saveData();
    setToastVisible(true);
  };

  return (
    <>
      <Button title="Save" onPress={handleSave} />
      <Toast
        visible={toastVisible}
        variant="success"
        message="Data saved successfully"
        onDismiss={() => setToastVisible(false)}
      />
    </>
  );
}
```
